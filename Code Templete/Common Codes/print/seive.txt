bool flag[SIZE_N+5];
int primes[SIZE_P+5];

int seive()
{
	int i,j,total=0,val;

	for(i=2;i<=SIZE_N;i++) flag[i]=1;

    	val=sqrt(SIZE_N)+1;

	for(i=2;i<val;i++)
       	if(flag[i])
            		for(j=i;j*i<=SIZE_N;j++)
                		flag[i*j]=0;

	for(i=2;i<=SIZE_N;i++)
       	if(flag[i])
            		primes[total++]=i;

    	return total;
}
